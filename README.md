    # FizzBuz:)

    while i < 16 {
        if i % 3 == 0 && i % 5 == 0{
            print("Fizz buz");
        }
        else if i % 3 == 0{
            print("Fizz");
        }
        else if i % 5 == 0{
            print("Buzz");
        }
        else{
            print(i);
        };

        i = i + 1;
    }
